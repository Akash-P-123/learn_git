# How to clone a repo
git clone "URL"

# How to check status , commit , add and push your repo in your git hub
git status
git add
git commit -m "Feat: comment " 
git commit --amend # Used for to edit the last commit in a test editor and also we can able to use the same commit message again 
git push
git push -f #force the push operation

#How to rename file_name
git mv old_file_name new_file_name 

#How to check your No. Of Branches in your repo 
git branch -a
git chekout branch_name

# How to  delete a branch 
git branch -d branch_name

# To delete in github interface/isolated branch we use 
git push origin --delete branch_name

# To delete a file in the branch
git rm file_name
git commit -m "Feat: comments"
git push origin main

#The Git reset Command
git reset --medium hash value 
git reset --soft hash value
git reset --hard hash value
